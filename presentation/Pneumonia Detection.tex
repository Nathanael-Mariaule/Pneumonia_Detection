\begin{frame}
The goal of the project is to build a machine learning model that diagnoses pneumonia based on a chest X-ray.


\end{frame}

\begin{frame}{Preprocession of the images}

\begin{itemize}
\item Resize (150,150)
\item 1 channel for grayscale
\item values between 0-1
\item Random data augmentation (rotation, flip, zoom and shift)
\end{itemize}
\end{frame}
